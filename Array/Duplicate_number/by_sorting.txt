#include <iostream>
#include <vector>
#include <bits/stdc++.h>

using namespace std;


void dup(vector<int>&vec,int n,int c){
	sort(vec.begin(),vec.end());
	for(int i=0;i<n;i++){
		if(vec[i]=vec[i+1]){
			c=vec[i];
			break;
		}
	}
	cout<<c<<endl;
}

int main(){
	int n,c;
	cin>>n;
	vector<int>vec;
	for(int i=0;i<n;i++){
		int inp;
		cin>>inp;
		vec.push_back(inp);
	}
	dup(vec,n,c);
}
